Prüfer code

In combinatorial mathematics, the Prüfer sequence (also Prüfer code or Prüfer numbers) of a labeled tree is a unique sequence associated with the tree. The sequence for a tree on n vertices has length n − 2, and can be generated by a simple iterative algorithm. 
            vector> adj;

                vector pruefer_code() {
                    int n = adj.size();
                    set leafs;
                    vector degree(n);
                    vector killed(n, false);
                    for (int i = 0; i < n; i++) {
                        degree[i] = adj[i].size();
                        if (degree[i] == 1)
                            leafs.insert(i);
                    }
                
                    vector code(n - 2);
                    for (int i = 0; i < n - 2; i++) {
                        int leaf = *leafs.begin();
                        leafs.erase(leafs.begin());
                        killed[leaf] = true;
                
                        int v;
                        for (int u : adj[leaf]) {
                            if (!killed[u])
                                v = u;
                        }
                
                        code[i] = v;
                        if (--degree[v] == 1)
                            leafs.insert(v);
                    }
                
                    return code;
                }
